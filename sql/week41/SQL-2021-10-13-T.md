FROM adress a
JOIN customer c ON c.adress.id = a.adress.id

```sql
SELECT p.payment_id, p.customer_id, p.amount * 0.05, r.return_date, p.payment_date
FROM payment p
JOIN rental r ON p.rental_id = r.rental_id
WHERE p.payment_date > r.return_date
```
